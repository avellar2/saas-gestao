# Gestor Local — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a multi-tenant modular SaaS platform (Gestor Local) for small Brazilian businesses, with company isolation, module activation by subscription, admin panel, client CRUD, quotes, service orders, PDF generation, and WhatsApp integration.

**Architecture:** Next.js 15 full-stack monolith with shared PostgreSQL schema + Row Level Security + Prisma extension for tenant isolation. Three-layer protection: Next.js middleware, Prisma extension, PostgreSQL RLS.

**Tech Stack:** Next.js 15, TypeScript, Prisma, PostgreSQL 16, NextAuth.js v5, Tailwind CSS, shadcn/ui, @react-pdf/renderer, Docker Compose

```typescript
// src/lib/prisma.ts
import { PrismaClient } from "@prisma/client";

const globalForPrisma = globalThis as unknown as {
  prisma: PrismaClient | undefined;
};

// Tables that have companyId for tenant filtering
const TENANT_MODELS = new Set([
  "User",
  "Customer",
  "Quote",
  "QuoteItem",
  "ServiceOrder",
  "ServiceOrderItem",
  "CompanyModule",
  "Subscription",
]);

function createPrismaClient() {
  return new PrismaClient();
}

// Base client (no tenant filtering — for auth, admin, and login)
export const prisma = globalForPrisma.prisma ?? createPrismaClient();

if (process.env.NODE_ENV !== "production") {
  globalForPrisma.prisma = prisma;
}

// Tenant-scoped client factory
// Usage: const db = tenantPrisma(companyId);
// Then use db.customer.findMany({}) — companyId is injected automatically
export function tenantPrisma(companyId: string) {
  return prisma.$extends({
    name: "tenant",
    query: {
      $allModels: {
        async $allOperations({ args, model, operation, query }) {
          if (!TENANT_MODELS.has(model)) {
            return query(args);
          }

          // For read operations, inject companyId into where
          if (operation.startsWith("find") || operation === "aggregate" || operation === "count") {
            args.where = { ...args.where, companyId };
          }

          // For create, inject companyId into data
          if (operation === "create") {
            args.data = { ...args.data, companyId };
          }

          // For createMany, inject companyId into each item
          if (operation === "createMany") {
            if (Array.isArray(args.data)) {
              args.data = args.data.map((item: Record<string, unknown>) => ({
                ...item,
                companyId,
              }));
            } else {
              args.data = { ...args.data, companyId };
            }
          }

          // For update/updateMany, ensure where still has companyId
          if (operation.startsWith("update") || operation === "delete" || operation === "deleteMany") {
            args.where = { ...args.where, companyId };
          }

          return query(args);
        },
      },
    },
  });
}

export type TenantPrismaClient = ReturnType<typeof tenantPrisma>;
```

// src/lib/auth.ts
import NextAuth from "next-auth";
import Credentials from "next-auth/providers/credentials";
import bcrypt from "bcryptjs";
import { prisma } from "./prisma";

export const { handlers, signIn, signOut, auth } = NextAuth({
  providers: [
    Credentials({
      name: "credentials",
      credentials: {
        email: { label: "E-mail", type: "email" },
        password: { label: "Senha", type: "password" },
      },
      async authorize(credentials) {
        if (!credentials?.email || !credentials?.password) {
          return null;
        }

        const user = await prisma.user.findUnique({
          where: { email: credentials.email as string },
          include: { company: true },
        });

        if (!user || !user.active) {
          return null;
        }

        if (user.company.status === "SUSPENDED" || user.company.status === "CANCELLED") {
          return null;
        }

        const isValid = await bcrypt.compare(
          credentials.password as string,
          user.passwordHash
        );

        if (!isValid) {
          return null;
        }

        return {
          id: user.id,
          email: user.email,
          name: user.name,
          role: user.role,
          companyId: user.companyId,
          companyStatus: user.company.status,
        };
      },
    }),
  ],
  callbacks: {
    async jwt({ token, user }) {
      if (user) {
        token.id = user.id;
        token.role = (user as Record<string, unknown>).role;
        token.companyId = (user as Record<string, unknown>).companyId;
        token.companyStatus = (user as Record<string, unknown>).companyStatus;
      }
      return token;
    },
    async session({ session, token }) {
      if (session.user) {
        session.user.id = token.id as string;
        (session.user as Record<string, unknown>).role = token.role;
        (session.user as Record<string, unknown>).companyId = token.companyId;
        (session.user as Record<string, unknown>).companyStatus = token.companyStatus;
      }
      return session;
    },
  },
  pages: {
    signIn: "/login",
  },
  session: {
    strategy: "jwt",
  },
});
```

const BASE_PRICE = 49.0;
const MODULE_PRICES = [50, 30, 25]; // 1st, 2nd, 3rd module
const ADDITIONAL_PRICE = 20; // 4th module onwards

export function calculateMonthlyPrice(activeModulesCount: number): number {
  if (activeModulesCount === 0) return BASE_PRICE;

  let total = BASE_PRICE;
  for (let i = 0; i < activeModulesCount; i++) {
    if (i < MODULE_PRICES.length) {
      total += MODULE_PRICES[i];
    } else {
      total += ADDITIONAL_PRICE;
    }
  }
  return total;
}

// src/middleware.ts
import { NextResponse } from "next/server";
import { auth } from "@/lib/auth";
import { getModuleKeyForPath, isModuleActive } from "@/lib/module-guard";

export default auth(async function middleware(req) {
  const { pathname } = req.nextUrl;

  // Allow auth routes and static files
  if (
    pathname.startsWith("/login") ||
    pathname.startsWith("/api/auth") ||
    pathname.startsWith("/_next") ||
    pathname.startsWith("/favicon")
  ) {
    return NextResponse.next();
  }

  // Check authentication
  const session = req.auth;
  if (!session?.user) {
    const loginUrl = new URL("/login", req.url);
    return NextResponse.redirect(loginUrl);
  }

  const role = (session.user as Record<string, unknown>).role as string;
  const companyId = (session.user as Record<string, unknown>).companyId as string;

  // Admin routes — only SUPER_ADMIN
  if (pathname.startsWith("/admin")) {
    if (role !== "SUPER_ADMIN") {
      return NextResponse.redirect(new URL("/dashboard", req.url));
    }
    return NextResponse.next();
  }

  // Dashboard routes — check module access
  if (pathname.startsWith("/dashboard") || pathname.startsWith("/clientes") || pathname.startsWith("/orcamentos") || pathname.startsWith("/ordens-servico")) {
    if (role === "SUPER_ADMIN") {
      return NextResponse.redirect(new URL("/admin", req.url));
    }

    // Check if company is active
    const companyStatus = (session.user as Record<string, unknown>).companyStatus as string;
    if (companyStatus === "SUSPENDED" || companyStatus === "CANCELLED") {
      return NextResponse.redirect(new URL("/login", req.url));
    }

    // Check module access
    const moduleKey = getModuleKeyForPath(pathname);
    if (moduleKey) {
      const hasAccess = await isModuleActive(companyId, moduleKey);
      if (!hasAccess) {
        const upgradeUrl = new URL("/upgrade", req.url);
        upgradeUrl.searchParams.set("module", moduleKey);
        return NextResponse.redirect(upgradeUrl);
      }
    }

    return NextResponse.next();
  }

  // Upgrade page — always accessible for logged-in non-admin users
  if (pathname.startsWith("/upgrade")) {
    return NextResponse.next();
  }

  return NextResponse.next();
});

export const config = {
  matcher: [
    "/((?!_next/static|_next/image|favicon.ico|robots.txt).*)",
  ],
};

import { NextRequest, NextResponse } from "next/server";
import { prisma } from "@/lib/prisma";
import { auth } from "@/lib/auth";
import { calculateMonthlyPrice } from "@/lib/pricing";

export async function GET(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  const session = await auth();
  if (!session || (session.user as Record<string, unknown>).role !== "SUPER_ADMIN") {
    return NextResponse.json({ error: "Não autorizado" }, { status: 401 });
  }

  const { id } = await params;
  const company = await prisma.company.findUnique({
    where: { id },
    include: {
      users: true,
      companyModules: { include: { module: true }, orderBy: { module: { sortOrder: "asc" } } },
      subscription: true,
    },
  });

  if (!company) {
    return NextResponse.json({ error: "Empresa não encontrada" }, { status: 404 });
  }

  // Calculate current monthly price
  const activeModulesCount = company.companyModules.filter((cm) => cm.active).length;
  const calculatedPrice = calculateMonthlyPrice(activeModulesCount);

  return NextResponse.json({ ...company, calculatedPrice });
}

export async function PUT(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  const session = await auth();
  if (!session || (session.user as Record<string, unknown>).role !== "SUPER_ADMIN") {
    return NextResponse.json({ error: "Não autorizado" }, { status: 401 });
  }

  const { id } = await params;
  const body = await req.json();

  // Update company basic data
  if (body.name || body.status) {
    await prisma.company.update({
      where: { id },
      data: {
        ...(body.name && { name: body.name }),
        ...(body.tradeName !== undefined && { tradeName: body.tradeName }),
        ...(body.document !== undefined && { document: body.document }),
        ...(body.phone !== undefined && { phone: body.phone }),
        ...(body.whatsapp !== undefined && { whatsapp: body.whatsapp }),
        ...(body.email !== undefined && { email: body.email }),
        ...(body.address !== undefined && { address: body.address }),
        ...(body.status && { status: body.status }),
      },
    });
  }

  // Toggle module
  if (body.moduleKey && body.moduleActive !== undefined) {
    const now = new Date();
    await prisma.companyModule.update({
      where: { companyId_moduleKey: { companyId: id, moduleKey: body.moduleKey } },
      data: {
        active: body.moduleActive,
        ...(body.moduleActive ? { activatedAt: now, deactivatedAt: null } : { deactivatedAt: now }),
      },
    });

    // Recalculate subscription
    const activeModules = await prisma.companyModule.count({
      where: { companyId: id, active: true },
    });
    const monthlyPrice = calculateMonthlyPrice(activeModules);

    await prisma.subscription.upsert({
      where: { companyId: id },
      update: {
        modulesCount: activeModules,
        monthlyPrice,
      },
      create: {
        companyId: id,
        modulesCount: activeModules,
        monthlyPrice,
        basePrice: 49.0,
      },
    });

    // Update company monthlyPrice
    await prisma.company.update({
      where: { id },
      data: { monthlyPrice },
    });
  }

  return NextResponse.json({ success: true });
}

  { value: "TRIAL", label: "Teste Grátis" },
  { value: "ACTIVE", label: "Ativa" },
  { value: "SUSPENDED", label: "Suspensa" },
  { value: "CANCELLED", label: "Cancelada" },
];

  const params = useParams();

  const [data, setData] = useState<CompanyDetail | null>(null);

  const [loading, setLoading] = useState(false);

  useEffect(() => {
    fetch(`/api/empresas/${params.id}`)
      .then((r) => r.json())
      .then((d) => {
        setData(d);
        setStatus(d.status);
      });
  }, [params.id]);

  if (!data) return <div>Carregando...</div>;

  async function toggleModule(moduleKey: string, active: boolean) {

      body: JSON.stringify({ moduleKey, moduleActive: !active }),
    });
    const res = await fetch(`/api/empresas/${params.id}`);
    const d = await res.json();
    setData(d);
    setLoading(false);
  }

  async function updateStatus() {
    setLoading(true);
    await fetch(`/api/empresas/${params.id}`, {
      method: "PUT",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ status }),

    setData(d);
    setStatus(d.status);
    setLoading(false);

            {data.phone && <div><span className="text-gray-500">Tel:</span> <span className="font-medium">{data.phone}</span></div>}
            {data.whatsapp && <div><span className="text-gray-500">WhatsApp:</span> <span className="font-medium">{data.whatsapp}</span></div>}

            >
              {STATUS_OPTIONS.map((opt) => (
                <option key={opt.value} value={opt.value}>{opt.label}</option>
              ))}
            </select>
            <Button size="sm" onClick={updateStatus} disabled={loading}>Salvar</Button>
          </div>
        </CardContent>
      </Card>

                <div>

                  </Button>
                </div>
              </div>
            ))}
          </div>
        </CardContent>
      </Card>

      <Card>
        <CardHeader><CardTitle>Usuários</CardTitle></CardHeader>
        <CardContent>
          <table className="w-full text-sm">
            <thead>

export async function GET(req: NextRequest) {
  const session = await auth();
  if (!session) return NextResponse.json({ error: "Não autorizado" }, { status: 401 });

  const companyId = (session.user as Record<string, unknown>).companyId as string;
  const companyStatus = (session.user as Record<string, unknown>).companyStatus as string;
  const db = tenantPrisma(companyId);

  const { searchParams } = new URL(req.url);
  const search = searchParams.get("search") || "";
  const page = parseInt(searchParams.get("page") || "1");
  const limit = 20;

  const where = search
    ? {
        OR: [
          { name: { contains: search, mode: "insensitive" as const } },
          { phone: { contains: search } },
          { whatsapp: { contains: search } },
        ],
      }
    : {};

  const [customers, total] = await Promise.all([
    db.customer.findMany({
      where,
      orderBy: { createdAt: "desc" },
      skip: (page - 1) * limit,
      take: limit,
    }),
    db.customer.count({ where }),
  ]);

  return NextResponse.json({ customers, total, page, totalPages: Math.ceil(total / limit) });
}

export async function POST(req: NextRequest) {
  const session = await auth();
  if (!session) return NextResponse.json({ error: "Não autorizado" }, { status: 401 });

  const companyId = (session.user as Record<string, unknown>).companyId as string;
  const companyStatus = (session.user as Record<string, unknown>).companyStatus as string;
  const db = tenantPrisma(companyId);

  // Check trial limits
  const customerCount = await db.customer.count();
  if (isTrialLimitReached(companyStatus, "customers", customerCount)) {
    return NextResponse.json(
      { error: "Limite de clientes do plano trial atingido (máximo 20)" },
      { status: 403 }
    );
  }

  const body = await req.json();
  const customer = await db.customer.create({
    data: {
      name: body.name,
      phone: body.phone,
      whatsapp: body.whatsapp,
      email: body.email,
      document: body.document,
      address: body.address,
      notes: body.notes,
    },
  });

  return NextResponse.json(customer, { status: 201 });
}

interface ClientFormProps {
  initialData?: {
    name: string;
    phone: string;
    whatsapp: string;
    email: string;
    document: string;
    address: string;
    notes: string;
  };
  onSubmit: (data: Record<string, string>) => Promise<void>;
  submitLabel?: string;
}

export function ClientForm({ initialData, onSubmit, submitLabel = "Salvar" }: ClientFormProps) {
  const [loading, setLoading] = useState(false);
  const [form, setForm] = useState({
    name: initialData?.name || "",
    phone: initialData?.phone || "",
    whatsapp: initialData?.whatsapp || "",

    document: initialData?.document || "",
    address: initialData?.address || "",
    notes: initialData?.notes || "",

    e.preventDefault();
    setLoading(true);
    await onSubmit(form);
    setLoading(false);

    setForm((prev) => ({ ...prev, [key]: value }));

import { NextRequest, NextResponse } from "next/server";
import { auth } from "@/lib/auth";
import { tenantPrisma, prisma } from "@/lib/prisma";
import { isTrialLimitReached } from "@/lib/company-limits";

export async function GET(req: NextRequest) {
  const session = await auth();
  if (!session) return NextResponse.json({ error: "Não autorizado" }, { status: 401 });

  const companyId = (session.user as Record<string, unknown>).companyId as string;
  const db = tenantPrisma(companyId);

  const { searchParams } = new URL(req.url);
  const status = searchParams.get("status") || undefined;

  const quotes = await db.quote.findMany({
    where: status ? { status: status as any } : undefined,
    include: { customer: { select: { name: true } }, items: true },
    orderBy: { createdAt: "desc" },
  });

  return NextResponse.json(quotes);
}

export async function POST(req: NextRequest) {
  const session = await auth();
  if (!session) return NextResponse.json({ error: "Não autorizado" }, { status: 401 });

  const companyId = (session.user as Record<string, unknown>).companyId as string;
  const companyStatus = (session.user as Record<string, unknown>).companyStatus as string;
  const db = tenantPrisma(companyId);

  const quoteCount = await db.quote.count();
  if (isTrialLimitReached(companyStatus, "quotes", quoteCount)) {
    return NextResponse.json({ error: "Limite de orçamentos do plano trial atingido" }, { status: 403 });
  }

  const body = await req.json();

  // Get next number for company
  const lastQuote = await db.quote.findFirst({
    orderBy: { number: "desc" },
    select: { number: true },
  });
  const nextNumber = (lastQuote?.number || 0) + 1;

  // Calculate totals
  const subtotal = body.items.reduce((sum: number, item: any) => sum + item.quantity * item.unitPrice, 0);
  const discount = body.discount || 0;
  const total = subtotal - discount;

  const quote = await db.quote.create({
    data: {
      customerId: body.customerId,
      number: nextNumber,
      description: body.description,
      subtotal,
      discount,
      total,
      validUntil: body.validUntil ? new Date(body.validUntil) : null,
      notes: body.notes,
      items: {
        create: body.items.map((item: any) => ({
          description: item.description,
          quantity: item.quantity,
          unitPrice: item.unitPrice,
          total: item.quantity * item.unitPrice,
        })),
      },
    },
    include: { items: true },
  });

  return NextResponse.json(quote, { status: 201 });
}

export async function PUT(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  const session = await auth();
  if (!session) return NextResponse.json({ error: "Não autorizado" }, { status: 401 });

  const companyId = (session.user as Record<string, unknown>).companyId as string;
  const { id } = await params;
  const db = tenantPrisma(companyId);

  const body = await req.json();

  // Update status only
  if (body.status) {
    const quote = await db.quote.update({
      where: { id },
      data: { status: body.status },
    });
    return NextResponse.json(quote);
  }

  // Update full quote (only if DRAFT)
  const existing = await db.quote.findUnique({ where: { id } });
  if (!existing || existing.status !== "DRAFT") {
    return NextResponse.json({ error: "Orçamento não pode ser editado" }, { status: 400 });
  }

  const subtotal = body.items.reduce((sum: number, item: any) => sum + item.quantity * item.unitPrice, 0);
  const discount = body.discount || 0;
  const total = subtotal - discount;

  // Delete old items and create new ones
  await db.quoteItem.deleteMany({ where: { quoteId: id } });

  const quote = await db.quote.update({
    where: { id },
    data: {
      description: body.description,
      subtotal,
      discount,
      total,
      validUntil: body.validUntil ? new Date(body.validUntil) : null,
      notes: body.notes,
      items: {
        create: body.items.map((item: any) => ({
          description: item.description,
          quantity: item.quantity,
          unitPrice: item.unitPrice,
          total: item.quantity * item.unitPrice,

    include: { items: true },

import { useState } from "react";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select";
import { Plus, Trash2 } from "lucide-react";
import { formatCurrency } from "@/lib/utils";

interface QuoteItem {
  description: string;
  quantity: number;
  unitPrice: number;
}

interface CustomerOption {
  id: string;
  name: string;
}

interface QuoteFormProps {
  customers: CustomerOption[];
  initialData?: any;
  onSubmit: (data: any) => Promise<void>;
  submitLabel?: string;
  readOnly?: boolean;
}

export function QuoteForm({ customers, initialData, onSubmit, submitLabel = "Salvar", readOnly = false }: QuoteFormProps) {
  const [loading, setLoading] = useState(false);
  const [customerId, setCustomerId] = useState(initialData?.customerId || "");
  const [description, setDescription] = useState(initialData?.description || "");
  const [discount, setDiscount] = useState(initialData?.discount || 0);
  const [validUntil, setValidUntil] = useState(
    initialData?.validUntil ? new Date(initialData.validUntil).toISOString().split("T")[0] : ""
  );
  const [notes, setNotes] = useState(initialData?.notes || "");
  const [items, setItems] = useState<QuoteItem[]>(
    initialData?.items?.map((i: any) => ({
      description: i.description,
      quantity: Number(i.quantity),
      unitPrice: Number(i.unitPrice),
    })) || [{ description: "", quantity: 1, unitPrice: 0 }]
  );

  const subtotal = items.reduce((sum, i) => sum + i.quantity * i.unitPrice, 0);
  const total = subtotal - discount;

  function addItem() {
    setItems([...items, { description: "", quantity: 1, unitPrice: 0 }]);
  }

  function removeItem(index: number) {
    setItems(items.filter((_, i) => i !== index));
  }

  function updateItem(index: number, field: keyof QuoteItem, value: string | number) {
    const newItems = [...items];
    newItems[index] = { ...newItems[index], [field]: value };
    setItems(newItems);
  }

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    setLoading(true);
    await onSubmit({
      customerId,
      description,
      discount,
      validUntil,
      notes,
      items: items.filter((i) => i.description && i.unitPrice > 0),
    });
    setLoading(false);
  }

  return (
    <form onSubmit={handleSubmit} className="space-y-4">
      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
        <div className="space-y-2">
          <Label>Cliente *</Label>
          <Select value={customerId} onValueChange={setCustomerId} disabled={readOnly}>
            <SelectTrigger><SelectValue placeholder="Selecione o cliente" /></SelectTrigger>
            <SelectContent>
              {customers.map((c) => (
                <SelectItem key={c.id} value={c.id}>{c.name}</SelectItem>
              ))}
            </SelectContent>
          </Select>
        </div>
        <div className="space-y-2">
          <Label>Validade</Label>
          <Input type="date" value={validUntil} onChange={(e) => setValidUntil(e.target.value)} disabled={readOnly} />
        </div>
      </div>

      <div className="space-y-2">
        <Label>Descrição</Label>
        <Input value={description} onChange={(e) => setDescription(e.target.value)} disabled={readOnly} />
      </div>

      <div>
        <div className="flex items-center justify-between mb-2">
          <Label>Itens</Label>
          {!readOnly && <Button type="button" variant="outline" size="sm" onClick={addItem}><Plus className="h-4 w-4 mr-1" /> Item</Button>}
        </div>
        <div className="space-y-2">
          {items.map((item, i) => (
            <div key={i} className="flex gap-2 items-start bg-gray-50 p-3 rounded">
              <div className="flex-1">
                <Input placeholder="Descrição" value={item.description} onChange={(e) => updateItem(i, "description", e.target.value)} disabled={readOnly} />
              </div>
              <div className="w-20">
                <Input type="number" placeholder="Qtd" value={item.quantity} onChange={(e) => updateItem(i, "quantity", parseFloat(e.target.value) || 0)} disabled={readOnly} />
              </div>
              <div className="w-28">
                <Input type="number" placeholder="Preço" value={item.unitPrice} onChange={(e) => updateItem(i, "unitPrice", parseFloat(e.target.value) || 0)} disabled={readOnly} />
              </div>
              <div className="w-24 py-2 text-right text-sm">{formatCurrency(item.quantity * item.unitPrice)}</div>
              {!readOnly && (
                <Button type="button" variant="ghost" size="sm" onClick={() => removeItem(i)}>
                  <Trash2 className="h-4 w-4 text-red-500" />
                </Button>
              )}
            </div>
          ))}
        </div>
      </div>

      <div className="grid grid-cols-1 md:grid-cols-3 gap-4 bg-gray-50 p-4 rounded">
        <div>
          <Label>Subtotal</Label>
          <p className="text-lg font-medium">{formatCurrency(subtotal)}</p>
        </div>
        <div>
          <Label>Desconto</Label>
          <Input type="number" value={discount} onChange={(e) => setDiscount(parseFloat(e.target.value) || 0)} disabled={readOnly} />
        </div>
        <div>
          <Label>Total</Label>
          <p className="text-lg font-bold">{formatCurrency(total)}</p>
        </div>
      </div>

      <div className="space-y-2">
        <Label>Observações</Label>
        <Input value={notes} onChange={(e) => setNotes(e.target.value)} disabled={readOnly} />
      </div>

      {!readOnly && <Button type="submit" disabled={loading}>{loading ? "Salvando..." : submitLabel}</Button>}
    </form>
  );
}
```
